import math
def get_cube_volume(side=1):
    return side ** 3
def get_sphere_volume(radius=1):
    return (4/3) * math.pi * (radius ** 3)
def get_pyramid_volume(length=1, width=1, height=1):
    return (length * width * height) / 3
def get_cylinder_volume(radius=1, height=1):
    return math.pi * (radius ** 2) * height
shapes = {
    "куб": get_cube_volume,
    "сфера": get_sphere_volume,
    "пирамида": get_pyramid_volume,
    "цилиндр": get_cylinder_volume
}
shape_name = input("Введите фигуру (куб, сфера, пирамида, цилиндр): ").lower().strip()
print("Введите параметры через пробел (если оставить пустым, применятся значения 1):")
print("Для куба: сторона")
print("Для сферы: радиус")
print("Для пирамиды: длина ширина высота")
print("Для цилиндра: радиус высота")
params_input = input("> ").split()
params = list(map(float, params_input))
selected_function = shapes[shape_name]
result = selected_function(*params)
print(f"Объем фигуры {shape_name}: {result:.2f}")
